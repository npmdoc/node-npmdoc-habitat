    function habitat(prefix, defaults) {
  if (!(this instanceof habitat))
    return new habitat(prefix, defaults);
  if (prefix) {
    if (prefix.match(/[a-z]+[A-Z]/)) {
      prefix = fromCamelCase(prefix);
    }
    this.prefix = prefix.toUpperCase();
  }
  if (defaults)

`defaults` is an object representing the defaults if a key cannot be
found in the environment. This should be used sparingly.

```js
var env = new habitat('airsupport', { port: 1024 })
// will try the environment first, then fall back to 1024
var port = env.get('port');
```

function habitat.load (path)

  • description and source-code
    function load(path) {
  path = path || '.env';
  if (!fileExists(path))
    return false;
  var exports = fs.readFileSync(pathutil.resolve(path)).toString().trim()

  if (exports.indexOf('{') == 0) {
    try {
      var params = flatten(JSON.parse(exports));
      process.env = xtend(params, process.env)
      return habitat;
    } catch(e) {
      throw new Error('could not parse environment file, expected json')
    }
  }

  exports.split('\n').filter(function(param) {
    return !!param.match(/(.+?)=(.*)/);
  }).map(function (param) {
    var match = param.replace(/^export /i, '').match(/(.+?)=(.*)/);
    var key = match[1];
    var value = match[2];
    if ((match = value.match(/^(?:'|")(.*)(?:'|")$/)))
      value = match[1];
    return { key: key, value: value };
  }).forEach(function (param) {
    if (typeof process.env[param.key] === "undefined") {
      process.env[param.key] = param.value;
    }
  });

  return habitat;
}

## habitat.load([*pathToEnvFile*])
Try to load a set of environment variables from a file. This **will
not** override whatever is in the environment. This is a change from
Habitat v1.x's behaviour. This means that you can chain multiple

    function parse(thing) {
  var bool = /^(true|false)$/;

  if (bool.test(thing))
    return thing === 'true';
  if (thing == parseInt(thing, 10))
    return parseInt(thing, 10);
  if (thing == parseFloat(thing))
    return parseFloat(thing);

  try {
    return JSON.parse(thing);
  } catch(e) {}

  return thing;
}
